########## FICHIER DES MESSAGES D'ERREURS ##########
# SYNTAXE EBNF :                                   #
# ------------                                     #
# <fichier> -> { <ligne> }                         #
# <ligne>   -> '#' <commentaires> |                #
#              ( <numero> <libelle> )              #
# REMARQUES :                                      # 
# ---------                                        #
# Une ligne commencant par '#' est un commentaire  #
# Ne pas mettre de `blancs' en debut de ligne      #
# Ne pas mettre de `blancs' en fin de ligne        #
# Ne mettre qu'un seul blanc entre <numero> et     #
# <libelle>                                        #
# Laisser une ligne vide ou `blanche' a la         #
# fin du fichier                                   #
####################################################
#
